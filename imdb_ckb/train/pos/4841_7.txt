لە کۆلۆنیەکان ئێمە هەموومان ئەوەندە ئاشنا نین بە ئارسەر ئاسکی، بۆیە نزیک بوو ئەم فیلمەم بەجێبهێڵم (کە بەم دواییە پێشبینی TCM ی هەبوو) بەهۆی کۆمێنتە نەرێنییەکانی لێرە لەسەر دەرکەوتنی لە "شەمەندەفەری جنۆکە" -- کە چاوەڕوانم دەکرد بێت بە تەواوی بێزارکەر. لەبری ئەوە سەرسامییەکی خۆشم هەبوو کە خۆمم بینی کە بە شێوەیەکی دەنگی پێدەکەنم. لایەنە جەستەییەکانی کۆمیدیای ئاسکی و کاتی خۆی لە کاتی گەیاندنی هێڵێکدا ئەوە پیشان دەدات کە ئەگەر چارلی چاپلین و وودی ئالن منداڵێکیان هەبووبێت، چیت دەست دەکەوێت. هیچ بەراوردکردنی لەگەڵ بود ئەبۆت یان هیچ یەکێک لە دابینکەرانی ئاسایی دیکەی ڕیلیف کۆمیک نییە کە لە فیلمەکانی ئەم ژانرەدا دێنە ئاراوە. مرۆڤ دەتوانێت هەست بکات، جگە لەوەش، ئەو تەوەرەی کە askey بە کۆمیدیای بەریتانیەوە دەبەستێتەوە دوای ٣٠ ساڵ؛ لانی کەم لە ڕوانگەیەکی ئەمریکییەوە ڕوونە کە زیاتر هاوبەشی لەگەڵ تیپی مۆنتی پایتۆن هەیە وەک لە هەر یەکێک لە هاوتاکانی لێرە. سەبارەت بە باقی فیلمەکە -- تا زیاتر فیلمت بینیبێت، ئەگەری ئەوە زیاترە کە لە کۆتاییدا پێشبینی بکەیت، بەڵام هێشتا تەواو سەرگەرمی و کەشوهەوایە و شایەنی چاوەڕوانی دەرکەوتنی داهاتوویەتی.
