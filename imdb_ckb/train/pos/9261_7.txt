پێت وایە کەسی ئاسایی زیاتر/کەمتر لەبارەیەوە دەزانێت: سینەمای ئێران یان تۆپی پێی ئێران؟ ئەوەی جێگای سەرنجە، ئەو دووانە دێنە پێشەوەی نوێترین فیلمی سینەماکاری مشتومڕاوی ئێرانی جەعفەر پەنەهی بە ناوی 'Offside'، چیرۆکێک کە تۆپی پێ یان دەستگەیشتن بە تۆپی پێ وەک پاشبنەمایەک بۆ زنجیرەیەک دیمەن بەکاردەهێنێت کە لە دەوری مافی مرۆڤ دەسوڕێتەوە بۆ ئەوەی شتێک بکات یان بچێتە شوێنێک و یەکێک لە مافی تاک بۆ درێژکردنەوەی ئەو ئیحترامە ئەگەر و کەی حەز دەکات. شتە سەیرەکە ئەوەیە، تۆ لە ئۆفساید دوور ناکەیتەوە ئەگەر فێری تۆپی پێی ئێران بوویت یان زۆرت لەبارەی سینەمای ئێران (مەگەر پەروەردەیەک بێت لە شێوازی مشتومڕاوی پەنەهی)، بەڵام تۆ هەست بە ڕۆشنبیری دەکەیت، کە لانیکەم کەسێک لە ئێرانەوە هەیەتی پرسێکی مشتومڕاوی وەرگرتووە کە بە ڕوونی هێشتا زۆر لە کاردایە لە نەتەوەیەکی وەک ئێران، و ئامادەیە بیخاتە بەردەم بینەرێکی نێودەوڵەتی نەک ئیستغلالکردنی. دەتوانم مۆندیالی ٢٠٠٦ لەبیر بکەم، دڵنیام زۆر کەس دەتوانن. هاوین بوو دوای ساڵی سەرەتاییم لە زانکۆ و دوای ئەو سلۆگەی کە ساڵی یەکەمم بوو، هاوینێکی تۆپی پێی مۆندیال وەک تۆنیکێکی جوان و کاتیی بینی. پێویست ناکات بڵێم، من بە کردەوە هەموو یارییەکم بینی کە لە یەک کاتدا لەگەڵ ئەوانی تردا نیشانیان دەدەن لە کۆتایی قۆناغەکانی گروپەکاندا بەجۆرێک کە بۆ ئەوەی ڕووداوەکان نەبینرێت وەک لە ئیسپانیادا دەبینرێت '82 کاتێک وا بیردەکرایەوە کە ئەڵمانیای ڕۆژئاوا و نەمسا بۆ بەدەستهێنانی بۆ بەدەستهێنانی ئەنجام لە نێوان ئەوانەی کە دەیانبینن هەردووکیان لە ڕێگەی لەسەر حیسابی جەزائیر.هەر کەسێک لە بەریتانیا کە سەیری پەخشی بی بی سی بوو لە ئێران – مەکسیک یاری، هەردوو نەتەوە لە پاڵەوانێتییەکە، لەوانەیە ئەوە لەبیر بێت کە لە لوتکەی نمایشەکەدا، گاری لینکەر، پێشکەشکاری تۆپی پێی بی بی سی، کۆمێنتێکی سنید بۆ ناساندنی یارییەکە دەرکرد. شتێکی وەک: 'ئێمە بینیمان زەبەلاحەکان ئەڵمانیا، ئینگلتەرا، ئەرجەنتین و هۆڵەند یاری بەڵام ئێستا ئێمە دەگەینە بینینی.........مەکسیک و ئێران شەڕی لەسەر بکە'. جەختکردنەوە لەسەر ئەو دووانەی دوایی کە کەمتر بوون ڕوون بوو و ڕەنگە سەیرکردنی ئەوان کارێکی ڕۆژانە بووبێت. جێگەی شەرمەزارییە چونکە خەڵک هەن، و فیلمەکانی وەک ئۆفساید پێشکەشیان دەکەن، کە بەڕاستی ئامادەن ئێران ببینن کە یاری دەکات تا ئەو ڕادەیەی کە ئەوان دەخەنە مەترسییەوە بۆ خۆشگوزەرانییان بۆ بەدەستهێنانی و ئەوە جێگەی داخە کاتێک خەڵک لە پۆستێکی دەسەڵات 'کەمتەرخەم' کاتێک کەسانی تر هەن کە هەر کارێک بکەن بۆ ئەوەی چانسی ئەوەیان هەبێت کە بە پێچەوانەوە بەشداری بکەن. سەید نەیشن و ئەمریکا. لێرەدا فیلمێک دەربارەی یارییەکی مۆندیال کە ئێران تێیدا بەشدارە بەڵام ڕەنگە هەندێک یان لەڕاستیدا زۆربەی سینەماکارانی ئێران بە باشی هەڵیانبژاردبێت چیرۆکێک پێشکەش بکەن کە لە دەوری سەرکەوتنە بەناوبانگەکەی ئێران بەسەر ئەمریکادا دەسوڕێتەوە و ئەوە بۆ ئەوان چی دەگەیەنێت، دەرهێنەر پانای هەڵدەبژێرێت بۆ سەیرکردن لە چیدا لە پشت پەردەوە و چیرۆکێک لە ستاندەکانەوە دەخاتە ڕوو چونکە ژنان لە یارییەکانی تۆپی پێ قەدەغە دەکرێن و ڕێگەیان پێنادرێت ئیکستازی بژێوی بردنەوەی یارییەک بە گەورەیییەکی بێئەندازە بژین. دەتوانن فیلمێکی دەستی قورس لە خەیاڵتاندا بێت کە وردەکارییەکانی 'سەرکەوتن'ی ئێران وەک فیلمێکی پڕوپاگەندەی داربڕین کە بۆ ئیستغلالکردن دیزاین کراوە وەک چۆن 'دەسەڵات' هێزێکی ڕۆژئاوایی یان 'دوژمن' لە وەرزشێکدا کە ڕەنگە چاوەڕوان نەدەکرا ئەوە بکەن. کاتێک باس لە لێدانی بێ ڕکابەر لە یارییەکەی خۆیاندا دەکرێت، فیلمی هیندی ساڵی ٢٠٠١ی 'لاگان'ە، کە تێیدا جووتیارە هیندییەکان لە کریکێتی حوکمڕانی بەریتانییەکان دەشکێنن. سیمیلالی، 'یاری ژیانیان' لە ساڵی ٢٠٠٥، ئەمریکییەکان تەنها ئەمجارەیان لە مۆندیالی ١٩٥٠دا لە یارییەکی تۆپی پێدا لە ئینگلیزەکان دەدۆڕێنێت. جیاوازی نێوان ئەم فیلمانە و ئۆفسایدەکان زۆر گەورەن؛ ئۆفساید سەیری دۆخێک دەکات لە پشت عەقڵیەتی ویلایەتێکەوە و بوێری ئەوەی هەیە کە وەک کچەکان جەوهەرێکی فێمینیزم بگەڕێت، هەوڵدەدات ئێرانی خۆشەویستیان بەحرەین بشکێنێت و سەرکەوێت بۆ مۆندیال، وەک ئاژەڵەکان کۆدەکرێنەوە و لە پێیەکانی تەنها پاککەرەوەی خامدا دەهێڵرێنەوە لە پەنجەرەیەکی باررێدەوە کە ڕێگەیان پێدەدا یارییەکە ببینن. ئیستدلالەکە پەیوەندی بەو زمانە ناشرینەوە هەیە کە پیاوەکان 'ڕەنگە'، نەک ئیرادە، بەڵام 'ڕەنگە' لە کاتی یارییەکەدا بەکاریبهێنن. فیلمەکە ئێران ئەوەندە بێتوانای پێشکەش دەکات کە ناتوانن مافی پیاوێک ببەن بۆ سوێندخواردن و قەدەغەکردنی زمانە خراپەکە بەڵام دەبێت مافی ژنێک قەدەغە بکات بۆ بینینی یارییەکە هەمووی پێکەوە – سەیر نییە فیلمەکەی پەنەهی خۆی لە ئێران قەدەغە کرابوو. بۆ ئەوەی ئەو یاسایە بگرێتەوە کە یاساکە لە پلەی یەکەمدا ئەوەندە شێت دەکات بە ڕێگەدان بە کارەکتەرە ژنەکانی کە هەم جگەرە بکێشن و هەم مۆبایل بەکاربهێنن، دوو شت کە پزیشک و زانایان هەیانبووە پێت وایە زۆر مەترسیدارترە بۆ جەستەی مرۆڤ کەمێک زمانی ناشرین لە یاری تۆپی پێدا. پەنەهی گرنگییەکی تایبەتی بە ناونیشانی فیلمەکەی 'Osside' دەدات کە خۆی حوکمێکە لەناو تۆپی پێدا وردەکاری کاتێک 'هێرشکارێک' 'گیراوە' کە هەوڵدەدات سوودێک بەدەستبهێنێت لە بواری یاریکردندا. هاوتەریبەکان لەگەڵ پێگەی ئافرەتان لە فیلمەکەدا بەڕێوەدەچن، چونکە ئەم تاکانەی کە پێشڕەوی دەکەن (کە گوایە بێ ڕاستن) دەگیرێن کە هەوڵدەدەن یەکێکیان بخەنە سەر تیمی ئۆپۆزسیۆن؛ کە خودی دەوڵەتە. پێموایە ئۆفساید یەکێکە لە فیلمە باشەکان کە لە ڕۆژهەڵاتی ناوەڕاست و ناوچەی کەنداوەوە دێتە دەرەوە کە من بینیومە. فیلمێکی گرژ بەڵام خۆبەزلزانینە دەربارەی ئەو کەسانەی کە بەپێی IMDB هیچ مێژوویەکی نواندن هەڵناگرن و بەم شێوەیە، نواندنەکە خام و ڕاستەقینە و زیاد دەکات کە ئاورایەکی نیوە ڕیاڵیزمی سەیر و سەمەرە بۆ هەموو ئەمانە. کۆتاییە دەستبەجێیەکە جێگەی سەرنجە و شەڕی زارەکی بەردەوامی نێوان سەربازە نێرەکان و ئەو ژنە هەڵەکارانەیە کە بوێری سەرپێچیکردنیان لە دەوڵەت کردووە، شتێکی نایاب و سەرنجڕاکێشە. هەروەها پێموایە بۆ هەر کەسێک کە بەڕاستی حەزی لە فیلمەکە نەبێت بەهۆی ئەو هۆکارانەی کە باسم کردووە ئاستێکی دیاریکراوی بێبەزەیی نیشان دەدات، یان ئەوە یان تۆ لە حکومەتی ئێران کار دەکەیت.
