من دەبێ دان بە نائومێدییەکی کەمدا بنێم لەم فیلمەدا؛ زۆرم خوێندبووەوە کە چەندە سەرنجڕاکێشە، بەڵام لەگەڵ ئەوەشدا زنجیرە ئایندەییەکانی ڕاستەقینە، سەردەمی زانست، بڕێکی زۆر کەمی فیلمەکە دەگرێتەوە. سێتەکان و نایاب کاتێک دەگەینە ئەوان، و هەندێک وێنەی سەرسوڕهێنەر هەیە، بەڵام ئەم زنجیرە کۆتاییە کەمیی هەیە لە زۆری تردا...زۆر باشترین درامای پارچەکە لە بەشی ناوەڕاستدایە، و دواتر یاری دەکات وەک میلۆدراما، کە لە سروشتی زانستی خەیاڵی 'چەمکی بەرز'ی هەموو ئەمانەەوە سەرچاوە دەگرێت، و گفتوگۆیەکی بەهێزی تەواو نییە. ژیانی مرۆیی زۆر زیاتر لەم بەشەدا هەیە، بەڵام لەگەڵ ڕالف ڕیچاردسۆنی گەورە کە بە شکۆمەندی بەسەر لوتکەدا دەڕۆیشت وەک دیکتاتۆری بچووک، سەرۆکی هەموو شارەکان. من زۆرم خۆشدەویست ڕیچاردسۆن شێوازگەرایی و گەیاندنی کورتی هێڵەکان، ڕەتکردنەوەی ئامادەبوون و بیرۆکەکانی سەردانیکەر و متمانە بەخۆبوونی ڕیمۆند ماسی. ئەم سەرۆکە کەسایەتییەکی پۆستۆرە، بە شێوەیەکی قەناعەت پێکەر وەهمدارە، ناتوانێت سروشتی بچووکی شانشینی خۆی بەدی بهێنێت... ڕۆڵێکی گەورە نییە، بەڵام ڕیچاردسۆن زۆری لێ دروست دەکات. پێدەچێت هەموو شارێک مەبەست لێی ئینگلتەرا بێت، یان لانیکەم شارۆچکەیەکی ئینگلیزی تا ڕادەیەک نوێنەری ئینگلتەرا بێت. سەرنجڕاکێش بوو، خۆپاراستنی تەواو لە هەر لایەنێکی ئایینی بۆ شتەکان؛ پێدەچێت 'شتەکانی داهاتوو' لە دەوری ململانێیەکی نێوان بەربەریی جەنگی و باوەڕێک بە زانست بسووڕێنەوە کە پێدەچێت ئامانجی کۆتایی کەمی هەبێت، بەڵام تەنها بەردەوام بێت و بەردەوام بێت. هەوڵێکی دواکەوتوو هەیە بۆ وروژاندنی هەندێک ئارگیومێنت و گرژی لە بەشی کۆتاییدا، سەبارەت بە 'ژیانێکی' کەسی زیاتر، بەڵام یەکێکیان تەواو ناڕازی دەمێنێتەوە. فیلمەکە زۆر گرنگی بە ئاڵۆزییە وردەکان نەداوە؛ دەچێت بۆ بارانستۆرمینگ بینین و نادیار و ئەخلاقێکی بێدەنگ، هەموو جارێک. وە، بێگومان، بیری کۆتایی گرەواوی گرەواوی: ڕیمۆند ماسی کە بەش دەکات لیریکی سەبارەت بەوەی کە شتەکان چەندە نادیارن! سەبارەت بە پرسی فیلمەکە پێشبینیکردنە: دەبێ بڵێم بە هیچ شێوەیەک خراپ نییە وەک خۆی، بە لەبەرچاوگرتنی ئەوەی کە مرۆڤ بە ئاشکرا ڕێگە دەدات کە مەحاڵە وردەکارییەکانی ژیان هەر شتێکی وەک ڕاست بەدەست بهێنرێت. چەمکە گەورەکان شتێکیان هەیە بۆ ئەوان؛ شەڕێک لە ساڵی ١٩٤٠، باشە کە ڕەنگە پێشبینیکراو بووبێت... نزیکەی ٣٠ ساڵ دەخایەنێت، بیر!؟ بۆمبێکی ئەتۆمی - دەمانچەی سەروو یان هەندێک لەو جۆرە کەش و هەوایە - لە ساڵی ٢٠٣٦... سۆسیالیستی تەکنۆکراتی ئێمە باوەڕمان بە حکومەتی جۆری دەوڵەتە نەتەوەییە سەربەخۆکان نییە، لە بەریتانیا، دوای ساڵی ١٩٧٠ چاپەکان! ;-) هیچ سیاسەتێکی ڕاستەقینە لێرەدا نەچووە ناو کە جێگەی شەرمەزارییە؛ هەموو ئەو ڕووکارانە هەستێکی دژە جەنگی زۆر جێگەی ستایشە. بەگشتی، وا گریمان دەکرێت کە دیکتاتۆریەت - چ سەر ئێسک-لودی-فاشیست، وەک لە ژێر سەرۆک، یان هەموو دەستەکان-بۆ-پامپی سۆسیالیزمی زانستی - دەبێت *ببنە گرێبەست*، و ئەم کاریگەرییانە ناکرێت... لە کاتێکدا کە دەبێ ئەوەمان لەبیربێت کە لە ساڵی ١٩٣٦دا، هیچ زانیارییەک بە هیچ شێوەیەک نەبوو کە نازیزم و کۆمۆنیزم چۆن دەربچێت - یان تەنانەت چۆن دەرکەوتن - نەبوونی ڕەچاوکردنی ئەمە پێدەچێت نەرم و نیان بێت لە تەنیشت بازنەی دیدگای سینەماکاران لەسەر بابەتە کانی تر .زۆرێک لە شتەکانی پێشووتر دەبێ - و دەکرا - بە بۆچوونی من بڕدرابن؛ تەنها کورتترین شتەکانی '1940' پێویست دەبوو، بەڵام ئەم بەشە مەیلی ئەوەی هەیە کە تا ڕادەیەک قورس بێت، و ئەوە تەمەنەکان پێش ئەوەی بگەینە بەشەکانی ڕیچاردسۆن-ماسی. حەزم دەکرد زیاتر لەگەڵ مارگاریتا سکۆت ببینم؛ کە تەنها گوماناوییەکی بچووکە، فیگەرێکی چاوی دەریای ناوەڕاست دەبڕێت بۆ مەبەستێکی کەم. کارەکتەرەکە لێکۆڵینەوەی لەسەر ناکرێت، یان بە ڕاشکاوی ڕوون کراوەتەوە یان دەقۆزرێتەوە، تەنها یەک دیمەن نەبێت کە من تێکی نادەم و پەیوەندییەکەی لەگەڵ سەرۆکەکەدا لێکۆڵینەوەی لەسەر نەکراوە؛ بەڵام دواتر ئەمە ساڵانی ١٩٣٠ بوو و ئەو کاتە شتێکی وەک سانسۆری دامەزراوەیی بەربڵاو هەبوو. ئێدوارد چاپمان لە دوو ڕۆڵی خۆیدا بە شێوەیەکی سووک پێکەنیناوییە؛ زیاتر لە یەکەمدا وەکو چەپەڵێکی بەدبەخت، دوعاکردن بۆ جەنگ، تەنها بۆ ئەوەی بەڕاشکاوی لەلایەن کارەکتەری تری ماسییەوە دابنرێت. ماسی خۆی زۆر یارمەتی شتەکان دەدات، بەشەکانی خۆی بە تێکەڵەیەک لە خۆگرتن و گەمژەیی تاریک دەژەنێت، بە پێچەوانەوە بە باشی لەگەڵ کاستێکی تا ڕادەیەکی زۆر جیاوازدا، جگە لە ڕیچاردسۆن، و سکۆت و چاپمان، کەمێک. فیلمێک کە لە ساڵی ١٩٣٦ لە بەریتانیا دروستکراوە؛ یەکێک لەو چەند فیلمە زانستییە خەیاڵییە جددیانەی بەریتانی تا ئێستا، بەڕاستی! دیزاینەکەی (پارچە) و بەکارهێنانی سەرچاوەکانی، شتێکی زۆر گەورە و سەرسوڕهێنەرە. بەڵام، سیناریۆکە لە کۆتاییدا زیادەڕەوی دەکات و، لە هەندێک کاتدا، لە هەموو شوێنێک. ئاراستەکە تووشی تەختییەک دەبێت، هەرچەندە لە بۆنەیەکدا یەک دوو گێڕی دیمەندار بەرز دەکاتەوە. سینەماکار و بەڕێز ڕیچاردسۆن بەڕاستی شتەکان ڕزگار دەکەن بەڵام؛ بە رێککەوت دروستکردنی هەستێکی سەرسوڕهێنەر لە سەرسوڕمان لە تەکنەلۆژیا، و ئەنجامێکی سەرنجڕاکێش و جەرکی کە بەردەوام فریودەر. وەها شەرمەزارییەك ئەوەندە كەمێكی كەم یان چەمکی سینەمایی ڕاستەقینە هەیە بۆ هەموو شتەكە؛ پاول و پرێسبێرگەر دەبووە دەرهێنەری تەواو بۆ وەرگرتنی ئەرکێکی وەک ئەم - ئەوان بێ هاوتەمەن لە نێو دەرهێنەرانی بەریتانی وەک چیرۆکنووسانی بینراوی بوێر، کۆنتڕۆڵی گەورەی کارەکتەرەکان و بریکارەکان لە دیالۆگی ڕێژە یەکەم. ، سەیرێکی سەرنجڕاکێشە، شایەنی ئەوەن کە بخوێننەوە، بەڵام زۆر لە شارە گەورەکان کەمترە... 'tis زۆر وەک گەمژەیی، لە وێڵز'، وەک ئەو فیلمە لانگ، بەڵام لەگەڵ هیچ شتێک وەک هێزی سەرسوڕهێنەری ئەو.
