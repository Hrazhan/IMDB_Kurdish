من باوەڕم بەوە نییە کە تا ئێستا بەرنامەیەکی تەلەفزیۆنی خراپتر یان خراپتر هەبووە بۆ پەخشکردن لە ئەمریکا وەک یانەی ٧٠٠. ئەوان ئەمڕۆ هاوتای کو کلۆکس کلان لە سەدەی بیستەمدا. ڕق و کینەیان لە هەموو ئەو شتانەی کە باش و شیرین و مرۆڤ و پاکن لە هەموو توانای تێگەیشتنە. هێرشە بەردەوامەکانی ڕۆژانەیان بۆ سەر ملیۆنان و ملیۆنان ئەمریکی، هەروەها ملیارەها مرۆڤ لە سەرانسەری جیهان، کە بەڕێکەوت دیدگای خۆپەرست و دڕندانە و ئەهریمەنی و شێتانەیان بۆ مرۆڤایەتی هاوبەش نییە، لە دەرەوەی هەر شتێکە کە تا ئێستا تەلەفزیۆن بینیبێتی. ئەو درۆیانەی کە دەیڕژێنن و ئەو درۆ گاڵتەجاڕانەی کە هەوڵدەدەن وەک ڕاستی تێپەڕێنن، وەک بیرۆکەی "ژیان دوای مردن" یان "خودا" یان "گوناه" یان "شەیتان" ئەوەندە بێمانا و بێمانایە کە لە ڕاستیدا نەخۆشن، ئەوەندە نەخۆشن، ئەوەندە نەخۆشن، ئەوەندە نەخۆشن، ئەوەندە نەخۆشن، ئەوەندە نەخۆشن، ئەوەندە نەخۆشن، ئەوەندە نەخۆشن، ئەوەندە نەخۆشن، ئەوەندە نەخۆشن، ئەوەندە نەخۆشن، ئەوەندە نەخۆشن، ئەوەندە نەخۆشن، ئەوەندە نەخۆشن، بۆیە نەخۆشە، ئەوەندە نەخۆشە. ونبوون لە خەیاڵیاندا. خەڵکی عەقڵانی دەزانن کە ئایین مادەیەکی هۆشبەرە و نابێت ڕێگە بە خۆیان بدەن گیرۆدەی ئەو جۆرە فانتازییە بن. بەڵام یانەی ٧٠٠ بە تەنیا لە پۆلێکدا دایە. بەڕاستی ئەوان کولتێکن. لە کاتێکدا من باوەڕم بە ئازادی قسەکردن هەیە، بەڵام ئەو شێوازەی کە ڕق و سۆز و درۆ و زانیاری و بیرۆکەی خەیاڵی لەم جۆرە بڵاودەکەنەوە، لە هەموو سنوورەکان لە دەرەوەی هەموو سنوورەکانە. هیوادارم ڕۆژێک دێت کۆمەڵەی دەروونی ئەمریکی دواجار لێکۆڵینەوە لەو کەسانە بکات کە بەم شێوەیە خۆیان دەشێوێنن، ئەو کەسانەی کە رێگە بە خۆیان دەدەن بەو قووڵییە لە خاکی خەیاڵی ئاییندا نوقم بن کە چیتر هیچ چەمکێکی راستەقینەی واقیعیان نییە بە هیچ شێوەیەک . مامەڵەکردن لەگەڵ ئەو جۆرە کەسانەی کە تووشی ئەم وڵاتە بوون، لەم وڵاتەدا زۆر پێویستە، چونکە زۆر کەس بە تەواوی مێشکیان لەدەستداوە بۆ فانتازیاکانی ئایین. 700 یانە، هێندەی تر ترسناکە چونکە بەرز دەبێتەوە بۆ پێناسەی یاسایی 'کلت' بەڵام بەهۆی سامانە فراوانەکەی 700 یانەکە (ڕۆژانە لە ملیۆنان ئەمریکی کە لە گرتنی فێڵبازانەی خۆیاندا قفڵکراون) ئەوان لە سەرووی یاسادان لەم وڵاتەدا . بۆ ئەوانەی فیلمی "ماتریکس"یان بینیوە دەزانن کە فیلمەکە مەتافۆرێک بوو بۆ ئایین لەسەر زەوی: ئەو خراپانەی کە لە لوتکەی هەریەکێک لەو ئایینانەدان کە ئەوانەیان لێ دەڕژێنە گیریان خواردووە و بە شێوەیەکی دڕندانە سووکایەتییان پێکردووە بۆ ئەوان مەبەستی خۆپەرستی خۆی، و ئەو ملیۆنان کەسەی کە لە خەوێکی مەرگدا ڕاگیراون و هێواش هێواش لە هێزی ژیانیان ڕژان، نوێنەرایەتی ئەو کەسە زۆرانە دەکەن کە سەر بە ئایینەکانن و هەموو توانای هەستکردنیان بە هەستکردن بەو شتانەی لەدەستداوە کە بەڕاستی لە دەوروبەریان ڕوودەدات.<br /> <br />لە کاتە کەمتر مەدەنییەکاندا، خەڵکی شارە باشەکە ئەهریمەنەکانی وەک ئەوانەی پەیوەست بوون بە یانەی 700 لە دەرەوەی شارۆچکەکە بە مەشخەڵ و یاریگای یاریگا بەڕێوە دەبرد. بەڵام لە جیهانی ئەمڕۆدا کە خەڵک هەموو بژاردەیەکیان لەدەستداوە لە هەڵبژاردنی تەلەفزیۆنەکانیان کە پێشکەشیان دەکرێت، ئێمە هیچ ڕێگەیەکمان نییە بۆ ڕزگاربوون لە تاعونی 700 یانە. دەستدرێژییەکی سۆزداری و دڕندەیی وەها دڕندانە بکەن، وەک پێشکەش دەکرێت لەلایەن 700 یانە هەموو ڕۆژێک لە هەموو وڵاتەکەمان، لە خۆیان و منداڵەکانیان.
