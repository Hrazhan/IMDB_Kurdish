یانەی 700 دەبێت تاکە بەرنامەی تەلەفزیۆنی تەکفیر بێت لە مێژووی خودی تەلەفزیۆندا. بۆ ئەوەی دۆخەکە خراپتر بێت، لە ساڵی ١٩٦٦ەوە لە پەخشدایە، ئەمەش ئەوە دەگەیەنێت کە ئەگەر ملیۆنان کەس نەبن، هەزاران کەس دەیکڕن و دەچنە ناو ڕق و سۆزییەوە. بە سەرۆکایەتی پات ڕۆبەرتسۆن، دامەزرێنەر و سەرۆکی هاوپەیمانی مەسیحی، یانە 700ەکە لە زانیاری هەڵە دەمانباتە سەر هەڵە تێگەیشتن، هەواڵی پەخشکردن وەک ئەوان حەز دەکەن بیری لێبکەنەوە و هەوڵبدەن بینەرانی خۆی قەناعەت پێبکەن کە هەموو کێشەکانی جیهان بۆ ئەوەیە تاوانباری هاوڕەگەزبازان، ویككان، ڕۆحانیی سەردەمی نوێ، موسڵمانان، جولەكە، هیندۆس، مەسیحیی غەیرە بنەڕەتی، دیموكرات، دایكی تاقانە، بیانی، فێمینیست، پەرەسەندنخوازان، ژینگەپارێزان، زانایانی ناسا و هەر كەسێكی تر كە هاوبەشی بۆچوونە ئایینییە کۆنەپەرستەكانیان نییە. لە ڕاستیدا باشترین هەواڵی ساختەیە لە دوای نمایشی ڕۆژانە یان بەشی نوێکردنەوەی کۆتایی هەفتە لە شەوی شەممە ڕاستەوخۆ، یان لە دوای فۆکس نیوزەوە، بۆ ئەو مەسەلەیە. بێگومان، پات هەمیشە ئەو کەسەیە کە هەریەک لە بڕیارەکان دەدات، هەرچییەک دێتە مێشکم دەڵێت و نەفرەتێک نادات کە ئەو کەسەی توڕە دەکات یان ئازاری دەدات. لەم نێوەندەدا، ئەو بەشداری خۆی لە خەبات بۆ گۆڕینی ئەمریکا بۆ دەوڵەتێکی پۆلیسی سەربازی بە هەبوونی هەڵەی ئایینی لە هەموو شتێکدا لە هەموو شتێکدا دەتوانێت و داوای بەخشین لە دوای یەک لە دوای یەک - کەمتر لە 100 دۆلاری کەم نییە بۆ ئەوەی ببێتە یەکێک لە ئەندام، بەڕێگەدا - بۆ دابینکردنی بودجەی کانەکانی ئەڵماسی ئەفریقی پات و کڕینی نەوت لەو کۆمپانیایانە کە لەلایەن حکومەتەوە سەرزەنشتیان دەکرد لە ڕابردوودا بەهۆی خراپ بەکارهێنانی ژینگەکە. نەخێر، ئیتر شتێک نییە کە پات هاوڕێی باش بوو لەگەڵ دیکتاتۆرە جینۆسایدەکانی زەیر و زیمبابۆی بۆ ئەوەی یارمەتی بدات لە بەدەستهێنانی ئەو جۆرە سامانە؛ هەمووی بۆ شکۆمەندی گەورەتری خودایە، نازانیت؟ و بێگومان، خانەخوێی یانەی 700 هەمیشە ئامادەن ئەو پیتانە بخوێننەوە کە لەلایەن بینەرانەوە نووسراون وەک ئەوەی حەز دەکەن بیخەنە ڕوو، بە ڕێکەوت هەریەکەیان بە هەمان شێوە تایپ کراون و هەموویان لەسەر هەمان ڕەنگی کاغەز لەلایەن بینەرانەوە کە گوایە لەلایەن ئەو نەهامەتییانەی کە لەلایەن میواندارییەکان (ئەوان بانگەشەی ئەوە دەکەن کە وشەی زانینیان بۆ دێت) بەڵام هەرگیز لە بەرنامەکەدا دەرناکەون بۆ ئەوەی بڵێن چی بەسەریان هاتووە. ڕاستگۆیانە چۆن کەسێک دەتوانێت نمایشێک بە جددی وەربگرێت لە کاتێکدا تۆمارێکی چەپڵەی خراپ بەکاردەهێنێت؟ پێویستە خەڵک بپرسێت بۆچی هیچ بینەرێکی ستۆدیۆیی نییە. ئەمە کابرایەکە کە لەگەڵ جێری فالوێڵ هاوڕا بوو کە هێرشە تیرۆریستییەکانی ١١ی ئەیلوولی ٢٠٠١ بۆ سەر ئەمریکا لە ئەنجامی سزای خوداوەندمان بووە بەهۆی قبوڵکردنی هاوڕەگەزبازی و فێمینیزم. ئیرۆنی، بە لەبەرچاوگرتنی ئەوەی کە پات دوو جار بە ئاشکرا ئاماژەی بە جێبەجێکردنی چەکی ئەتۆمی لە وەزارەتی دەرەوەی ئەمریکا کردووە؛ گومانم نییە کە ئەو سامانەی کە بەهۆی ئەو جۆرە لێدوانانەیەوە دەستگیر نەکرێ، ئەوە بوو کە وای لێکرد دەستگیر بکرێت. قسە زلەکانی دژی هاوڕەگەزبازان و دایکانی تاقانە و هەر ژمارەیەک لە پراکتیزە سێکسییەکانی کە بە گوناهباری دەزانێت سەرنجڕاکێشە، بە لەبەرچاوگرتنی ئەوەی کە لە کاتی جەنگی کۆریادا ناسراوە بە ژمارەیەکی زۆر لە قەحبەخانە. وەک لە کتێبی پیرۆزدا هاتووە، بەرهەمدار و زۆرە، بۆیە پیرۆزە، پات - سوپاس بۆ تۆ، ڕەنگە ژمارەیەک منداڵ هەبن کە لە دایکانی تاقانە کۆریا لەدایک بوون. پاشان بێگومان ئەو کاتە هەبوو کە داوای تیرۆرکردنی سەرۆکی ڤەنزوێلا هوگۆ چاوێزی کرد (نەک ئەوەی پیرۆز بێت، بەڵام هێشتا). ئاه، بەڵێ، و با بەم زووانە ئەو کاتە لەبیر نەکەین کە ئەم خاچپەرستە بۆ ژیانی مرۆڤ پشتگیری لە لەباربردنی زۆرەملێی چین دەکات لە چین. زۆر مەسیحی لەبارەی ئەوەوە، ئایا تۆ ناڵێیت؟ وە تەنها لە ئەگەری ئەوەی پات لەبیری کردبێت، من وتارە بچووکەکەیم لەبیر نەکردووە کە مەسیحییە ئینجیلییەکان ئەمڕۆ بە تەواوی وەک جولەکەکان لە ئەڵمانیای نازیدا مامەڵەیان لەگەڵ دەکرێت. ڕاستگۆیانە، بەراوردکردنی دۆخی نالەباری ئەو بە ترسناکییەکانی هۆلۆکۆست نزیکە لە لێخۆشبوون. قسەکردن لەسەر ئەوە، پێویستە باسی ئەوە بکەم کە چۆن بە ئاشکرا درۆی کردووە کە هاوڕەگەزبازی لە نێوان حیزبی نازیدا بەڕێکەوت لە هەوڵێکی بەزەییانە بۆ بێناوبانگکردنی هاوڕەگەزبازان؟ بێگومان مێژوو ئەوەمان پێدەڵێت کە نازییەکان بە هەمان شێوەی ئەوەی بەرامبەر جولەکەکان مامەڵەیان کردووە، بەرەو هاوڕەگەزبازان مامەڵەیان کردووە. پات ڕۆبەرتسۆن یەکێکە لە گەورەترین درۆزنەکانی مێژوو. ئەگەر پینۆکیۆ بووایە، لوتی لە زەویدا دەگەڕا. ئەمە بانگەوازی بەئاگابوونتانە، خەڵک؛ یانەی 700 یەکێکە لە پرۆگرامە هەرە خراپترینەکان ئەگەر ناڕێکترین بێت لە مێژووی تەلەفزیۆندا. ئەوە خراپەکارییە کە خۆی وەک چاکە دەڕەخسێنێت؛ جل و بەرگێکی گورگ لە مەڕدایە. پیسییەکی تەکفیرییە کە هەوڵدەدات پاک و جوان و خۆشەویست دەربکەوێت. بەڵگەیەکی زیندووە کە تێڕوانینە ئایینییە ڕقاوی و مەترسیدارەکان لە هەندێک گرووپی ڕۆژهەڵاتی ناوەڕاستدا قەتیس نەماوە. تەنانەت ئەوانەی کە لە باوەڕ و بڕوای مەسیحی نین دەزانن کە پێچەوانەی هەموو ئەو شتانەیە کە عیسا فێری کردووە و ئەگەر عیسا بۆ ئەم یانەیە دەرکەوێت، ئەوا ئەوانیش نایگرێتەوە. لەبری ئەوە، ئەو سزایان دەدا کاتێک فەریسییەکانی سەردەمی خۆی دەکرد و سەبەتەی پارەی تەلەتونەکانیان هەڵدەدایەوە وەک ئەوەی لەبەردەم کەنیسەکەدا لە سەردەمی خۆیدا دەیکرد. تەنها ئەوەندە دەتوانم بڵێم سوپاس بۆ خوا کە پات هیچ چانسێکی نەبوو بۆ ئەوەی ببێتە سەرۆک؛ ئەگەر ئەو بیکردایە، ئەوا ئەو پێشبینییەی ئارمەگدۆن دەبوو - و لە لای کوڕە باشەکان نەدەبوو.
