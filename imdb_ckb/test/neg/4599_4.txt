فیلمێکی بێزارکەر دەربارەی دۆڕاوێکی بەدبەخت...ئەوە فاکتۆتۆمە بە کورتی. مات دیلۆن ڕۆڵی هێنری چینسکی دەگێڕێت، ئەلتەر ئیگۆی نووسەر چارڵز بوکۆڤسکی لەسەر بنەمای ڕۆمانەکەی کە بنکەکەی لە سەر ڕۆمانەکەیە. لەگەڵ ناسینی چیناسکی لەوانەیە ئەو نووسەر بێت بەڵام بە دڵنیاییەوە نووسەرێکی سەرکەوتوو نییە. ئەو ژیان بەناو کارێکی بێگەردیدا بە ژیاندا دەسوڕێتەوە و لە دوای کارێکی دیکەشەوە دەردەکرێت و کەمێک گرنگی پێنادات. ئەو ڕاستییەی کە هەمیشە سەرخۆش بووە ڕەنگە پەیوەندی بە خەمخۆرییەکەیەوە نەبێت. ژنێک بە ناوی جان دەناسێت کە لیلی تایلۆر ڕۆڵی تێدا دەگێڕێت و پەیوەندییەک بەدوای خۆیدا دەهێنێت. چیناسکی دەچێتە ناو شوقەکەی یان و ئێستا لەبری ئەوەی بە خۆی سەرخۆش بێت دەتوانێت لەگەڵ کەسێکی تردا سەرخۆش بێت. کاتە خۆشەکان. لە کۆتاییدا بەس ئەم پەیوەندییەی هەبووە بۆیە چینسکی بە تەنیا هێرش دەکاتە سەر خۆی. بەداخەوە هێشتا ناتوانێت کارێک ڕابگرێت بۆیە زوو شکاوە. لەم خاڵەدا ماریسا تۆمێی خۆی نیشان دەدات و بە هۆکارێک کارەکتەری خۆی، لۆرا، بڕیار دەدات ئەم دۆڕاوانەی سەرخۆشی بەدبەخت ڕزگار بکات کە تەنانەت نایزانێت. زۆری نەخایاند هەندێک لە ناسیاوەکانی لۆرا دەناسین و فیلمەکە بەم کۆمەڵە تا ڕادەیەک ئیکلێکتییەدا دەڕۆن و دەچنە ناو جۆرێک لە جیهانی بیزارۆوە. زۆری نەخایاند تۆمێی و هاوڕێکانی لە وێنەکە دەرچوون و جارێکی تر لەگەڵ چینسکی و خواردنەوە و ژیانە بچووکە بەدبەختەکەی دەمێنینەوە. هەندێک کاتێکی تر هەیە لەگەڵ یان بەسەر دەچێت بەڵام زۆربەی کات تەنها کاتێک هەیە کە بەسەر دەبەیت بۆ ئەوەی دۆڕاوێکی سەرخۆش و بێکار بێت. و پاشان فیلمەکە کۆتایی هاتووە و ساتێک زۆر زوو نییە. 94 خولەک لە یەکدەنگی ڕەها و پێدەچێت بێ سنوور درێژتر بێت لەگەڵ درێژبوونەوەی فیلمەکە. هیچ ڕووینەداوە. هیچ کاتێک ڕوونادات. ئەگەر دەتەوێت کاتژمێرێک و نیوێک لە ژیانت بەسەر بەریت سەیری مات دیلۆن بکە خواردنەوە ئەوا ئەمە فیلمەکەیە بۆ تۆ. ئەگەر تۆ بەدوای چیرۆکێکی سەرنجڕاکێش، کارەکتەرە باش پەرەپێدراوەکان یان هەر بەهایەکی کات بەسەربردن دەگەڕێیت ئەوا تۆ هاتوویتە شوێنی هەڵە. نمایشی دیلۆن لە ڕاستیدا بە هیچ شێوەیەک خراپ نییە. زۆر بەداخەوە ئەو فیلمەی کە دەوری داوە لە ڕاستیدا تا ڕادەیەک خراپە.
